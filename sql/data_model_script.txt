departments
-
dept_no CHAR(4) PK
dept_name VARCHAR(255)

employees
-
emp_no INT PK
birth_date DATE
first_name VARCHAR(255)
last_name VARCHAR(255)
gender CHAR(1)
hire_date DATE

-
emp_no INT FK >- employees.emp_no
dept_no CHAR(4) FK >- departments.dept_no
from_date DATE
to_date DATE

-
